local Utility = {}

function Utility.GetRoot(model)
	if not model then return nil end
	return model:FindFirstChild("HumanoidRootPart") or model.PrimaryPart
end

function Utility.Distance(a, b)
	local rootA = Utility.GetRoot(a)
	local rootB = Utility.GetRoot(b)

	if not rootA or not rootB then
		return math.huge
	end

	return (rootA.Position - rootB.Position).Magnitude
end

function Utility.IsAlive(unit)
	if not unit then return false end

	local humanoid = unit:FindFirstChildOfClass("Humanoid")
	if humanoid then
		return humanoid.Health > 0
	end

	local health = unit:GetAttribute("Health")
	return health ~= nil and health > 0
end

function Utility.GetTeam(unit)
	if not unit then return nil end
	return unit:GetAttribute("Team")
end

function Utility.AreEnemies(a, b)
	local teamA = Utility.GetTeam(a)
	local teamB = Utility.GetTeam(b)

	if not teamA or not teamB then
		return false
	end

	return teamA ~= teamB
end

function Utility.SetHealth(unit, value)
	local maxHealth = unit:GetAttribute("MaxHealth") or value
	local newHealth = math.clamp(value, 0, maxHealth)
	unit:SetAttribute("Health", newHealth)

	local humanoid = unit:FindFirstChildOfClass("Humanoid")
	if humanoid then
		humanoid.MaxHealth = maxHealth
		humanoid.Health = newHealth
	end

	if newHealth <= 0 then
		unit:SetAttribute("Dead", true)
	end
end

return Utility

-- Brouillard de guerre logique.
-- Important :
-- Roblox ne fournit pas un fog of war MOBA complet prêt à l'emploi.
-- Cette version crée une table de vision côté serveur.
-- Le client peut ensuite cacher/montrer les modèles selon ce que le serveur autorise.

local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local SharedConstants = require(ReplicatedStorage.Modules.SharedConstants)
local Utility = require(script.Parent.Utility)

local VisionService = {}

VisionService.VisibleUnitsByTeam = {
	Blue = {},
	Red = {},
}

local function clearVisionTables()
	VisionService.VisibleUnitsByTeam.Blue = {}
	VisionService.VisibleUnitsByTeam.Red = {}
end

local function addVisible(teamName, unit)
	if not teamName or not unit then return end
	if not VisionService.VisibleUnitsByTeam[teamName] then return end

	VisionService.VisibleUnitsByTeam[teamName][unit] = true
end

local function revealAround(sourceUnit, radius)
	local sourceRoot = Utility.GetRoot(sourceUnit)
	if not sourceRoot then return end

	local sourceTeam = Utility.GetTeam(sourceUnit)
	if not sourceTeam then return end

	for _, candidate in ipairs(workspace.Units:GetChildren()) do
		local candidateRoot = Utility.GetRoot(candidate)
		if candidateRoot then
			local distance = (sourceRoot.Position - candidateRoot.Position).Magnitude
			if distance <= radius then
				addVisible(sourceTeam, candidate)
			end
		end
	end
end

function VisionService.CanTeamSeeUnit(teamName, unit)
	if not teamName or not unit then return false end

	if Utility.GetTeam(unit) == teamName then
		return true
	end

	local tableForTeam = VisionService.VisibleUnitsByTeam[teamName]
	if not tableForTeam then return false end

	return tableForTeam[unit] == true
end

function VisionService.CanPlayerSeeUnit(player, unit)
	local teamName = player:GetAttribute("Team")
	return VisionService.CanTeamSeeUnit(teamName, unit)
end

function VisionService.UpdateVision()
	clearVisionTables()

	for _, unit in ipairs(workspace.Units:GetChildren()) do
		if Utility.IsAlive(unit) then
			local unitType = unit:GetAttribute("UnitType")
			local radius = 0

			if unitType == "Champion" then
				radius = SharedConstants.VISION.PLAYER_RADIUS
			elseif unitType == "Minion" then
				radius = SharedConstants.VISION.MINION_RADIUS
			elseif unitType == "Tower" then
				radius = SharedConstants.VISION.TOWER_RADIUS
			elseif unitType == "JungleMonster" then
				radius = SharedConstants.VISION.JUNGLE_MONSTER_RADIUS
			end

			if radius > 0 then
				revealAround(unit, radius)
			end
		end
	end
end

function VisionService.Start()
	task.spawn(function()
		while true do
			VisionService.UpdateVision()
			task.wait(SharedConstants.VISION.UPDATE_INTERVAL)
		end
	end)
end

return VisionService

local ReplicatedStorage = game:GetService("ReplicatedStorage")

local SharedConstants = require(ReplicatedStorage.Modules.SharedConstants)
local Utility = require(script.Parent.Utility)
local VisionService = require(script.Parent.VisionService)

local CombatService = {}

local lastAttackByUnit = {}

local function getAttackCooldown(attacker)
	return attacker:GetAttribute("AttackCooldown") or SharedConstants.COMBAT.AUTO_ATTACK_COOLDOWN
end

local function getAttackRange(attacker)
	return attacker:GetAttribute("AttackRange") or SharedConstants.COMBAT.AUTO_ATTACK_RANGE
end

local function getAttackDamage(attacker)
	return attacker:GetAttribute("AttackDamage") or SharedConstants.COMBAT.BASE_ATTACK_DAMAGE
end

local function canAttack(attacker, target, player)
	if not Utility.IsAlive(attacker) then
		return false, "Attacker dead"
	end

	if not Utility.IsAlive(target) then
		return false, "Target dead"
	end

	if not Utility.AreEnemies(attacker, target) then
		return false, "Not enemies"
	end

	local distance = Utility.Distance(attacker, target)
	if distance > getAttackRange(attacker) then
		return false, "Out of range"
	end

	local now = os.clock()
	local lastAttack = lastAttackByUnit[attacker] or 0
	local cooldown = getAttackCooldown(attacker)

	if now - lastAttack < cooldown then
		return false, "Attack cooldown"
	end

	if player then
		if not VisionService.CanPlayerSeeUnit(player, target) then
			return false, "Target not visible"
		end
	end

	return true
end

function CombatService.ApplyDamage(attacker, target, amount)
	local health = target:GetAttribute("Health")
	if not health then return end

	local newHealth = health - amount
	Utility.SetHealth(target, newHealth)

	if newHealth <= 0 then
		CombatService.HandleDeath(attacker, target)
	end
end

function CombatService.HandleDeath(killer, victim)
	victim:SetAttribute("Dead", true)

	local humanoid = victim:FindFirstChildOfClass("Humanoid")
	if humanoid then
		humanoid.Health = 0
	end

	-- TODO :
	-- - donner gold/XP au killer
	-- - respawn champion
	-- - respawn jungle après timer
	-- - supprimer minions
end

function CombatService.TryAutoAttack(attacker, target, player)
	local ok, reason = canAttack(attacker, target, player)
	if not ok then
		return false, reason
	end

	lastAttackByUnit[attacker] = os.clock()

	local damage = getAttackDamage(attacker)
	CombatService.ApplyDamage(attacker, target, damage)

	return true, "Attack success"
end

return CombatService

local TeamService = {}

function TeamService.AssignTeam(player)
	local blueCount = 0
	local redCount = 0

	for _, other in ipairs(Players:GetPlayers()) do
		local teamName = other:GetAttribute("Team")
		if teamName == SharedConstants.TEAMS.BLUE then
			blueCount += 1
		elseif teamName == SharedConstants.TEAMS.RED then
			redCount += 1
		end
	end

	if blueCount <= redCount then
		player:SetAttribute("Team", SharedConstants.TEAMS.BLUE)
	else
		player:SetAttribute("Team", SharedConstants.TEAMS.RED)
	end
end

function TeamService.Start()
	Players.PlayerAdded:Connect(function(player)
		TeamService.AssignTeam(player)
	end)
end

return TeamService

-- Pré-requis dans Workspace :
-- Workspace.Map.Paths.Blue.Top
-- Workspace.Map.Paths.Blue.Mid
-- Workspace.Map.Paths.Blue.Bot
-- Workspace.Map.Paths.Red.Top
-- Workspace.Map.Paths.Red.Mid
-- Workspace.Map.Paths.Red.Bot
--
-- Chaque lane contient des Parts numérotées : 1, 2, 3, 4...
-- Les minions avancent de waypoint en waypoint.

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")

local SharedConstants = require(ReplicatedStorage.Modules.SharedConstants)
local MinionData = require(ReplicatedStorage.Modules.MinionData)
local CombatService = require(script.Parent.CombatService)
local Utility = require(script.Parent.Utility)

local MinionService = {}

local waveNumber = 0

local function getPath(teamName, laneName)
	return workspace.Map.Paths[teamName][laneName]:GetChildren()
end

local function sortWaypoints(waypoints)
	table.sort(waypoints, function(a, b)
		return tonumber(a.Name) < tonumber(b.Name)
	end)
	return waypoints
end

local function setupMinion(minion, teamName, minionType)
	local data = MinionData.Types[minionType]

	minion:SetAttribute("UnitType", "Minion")
	minion:SetAttribute("Team", teamName)
	minion:SetAttribute("MinionType", minionType)
	minion:SetAttribute("MaxHealth", data.MaxHealth)
	minion:SetAttribute("Health", data.MaxHealth)
	minion:SetAttribute("AttackDamage", data.AttackDamage)
	minion:SetAttribute("AttackRange", data.AttackRange)
	minion:SetAttribute("AttackCooldown", data.AttackCooldown)
	minion:SetAttribute("MoveSpeed", data.MoveSpeed)
	minion:SetAttribute("GoldReward", data.GoldReward)
	minion:SetAttribute("XPReward", data.XPReward)
	minion:SetAttribute("Dead", false)

	local humanoid = minion:FindFirstChildOfClass("Humanoid")
	if humanoid then
		humanoid.WalkSpeed = data.MoveSpeed
		humanoid.MaxHealth = data.MaxHealth
		humanoid.Health = data.MaxHealth
	end

	minion.Parent = workspace.Units
end

local function findNearestEnemy(minion)
	local bestTarget = nil
	local bestDistance = math.huge
	local attackRange = minion:GetAttribute("AttackRange") or 8

	for _, candidate in ipairs(workspace.Units:GetChildren()) do
		if Utility.IsAlive(candidate) and Utility.AreEnemies(minion, candidate) then
			local distance = Utility.Distance(minion, candidate)
			if distance < bestDistance and distance <= attackRange then
				bestTarget = candidate
				bestDistance = distance
			end
		end
	end

	return bestTarget
end

local function runMinionAI(minion, waypoints)
	task.spawn(function()
		local humanoid = minion:FindFirstChildOfClass("Humanoid")
		if not humanoid then return end

		local waypointIndex = 1

		while Utility.IsAlive(minion) do
			local target = findNearestEnemy(minion)

			if target then
				CombatService.TryAutoAttack(minion, target, nil)
				task.wait(0.2)
			else
				local waypoint = waypoints[waypointIndex]
				if waypoint then
					humanoid:MoveTo(waypoint.Position)
					humanoid.MoveToFinished:Wait(2)
					waypointIndex += 1
				else
					task.wait(0.5)
				end
			end
		end

		task.wait(3)
		if minion then
			minion:Destroy()
		end
	end)
end

local function spawnSingleMinion(teamName, laneName, minionType, spawnPosition)
	local template = ServerStorage.Minions:FindFirstChild(minionType)
	if not template then
		warn("Missing minion template:", minionType)
		return
	end

	local minion = template:Clone()
	minion:PivotTo(CFrame.new(spawnPosition))
	setupMinion(minion, teamName, minionType)

	local waypoints = sortWaypoints(getPath(teamName, laneName))
	runMinionAI(minion, waypoints)
end

function MinionService.SpawnWaveForTeamAndLane(teamName, laneName)
	local spawnPoint = workspace.Map.MinionSpawns[teamName][laneName]
	local basePosition = spawnPoint.Position

	for i = 1, MinionData.Wave.MeleeCount do
		spawnSingleMinion(teamName, laneName, "Melee", basePosition + Vector3.new(i * 3, 0, 0))
	end

	for i = 1, MinionData.Wave.RangedCount do
		spawnSingleMinion(teamName, laneName, "Ranged", basePosition + Vector3.new(i * 3, 0, 5))
	end

	if waveNumber % MinionData.Wave.SiegeEveryNWaves == 0 then
		spawnSingleMinion(teamName, laneName, "Siege", basePosition + Vector3.new(0, 0, 10))
	end
end

function MinionService.SpawnWave()
	waveNumber += 1

	for _, teamName in pairs(SharedConstants.TEAMS) do
		for _, laneName in pairs(SharedConstants.LANES) do
			MinionService.SpawnWaveForTeamAndLane(teamName, laneName)
		end
	end
end

function MinionService.Start()
	task.spawn(function()
		while true do
			MinionService.SpawnWave()
			task.wait(SharedConstants.MATCH.MINION_WAVE_INTERVAL)
		end
	end)
end

return MinionService

local ReplicatedStorage = game:GetService("ReplicatedStorage")

local CombatService = require(script.Parent.CombatService)
local Utility = require(script.Parent.Utility)

local TowerService = {}

local function findTowerTarget(tower)
	local bestTarget = nil
	local bestDistance = math.huge
	local range = tower:GetAttribute("AttackRange") or 90

	for _, candidate in ipairs(workspace.Units:GetChildren()) do
		if Utility.IsAlive(candidate) and Utility.AreEnemies(tower, candidate) then
			local distance = Utility.Distance(tower, candidate)
			if distance <= range and distance < bestDistance then
				bestTarget = candidate
				bestDistance = distance
			end
		end
	end

	return bestTarget
end

local function setupTower(tower)
	tower:SetAttribute("UnitType", "Tower")
	tower:SetAttribute("MaxHealth", tower:GetAttribute("MaxHealth") or 3000)
	tower:SetAttribute("Health", tower:GetAttribute("Health") or 3000)
	tower:SetAttribute("AttackDamage", tower:GetAttribute("AttackDamage") or 150)
	tower:SetAttribute("AttackRange", tower:GetAttribute("AttackRange") or 90)
	tower:SetAttribute("AttackCooldown", tower:GetAttribute("AttackCooldown") or 1.0)
	tower:SetAttribute("Dead", false)
	tower.Parent = workspace.Units
end

function TowerService.Start()
	for _, tower in ipairs(workspace.Map.Towers:GetChildren()) do
		setupTower(tower)

		task.spawn(function()
			while tower and tower.Parent and Utility.IsAlive(tower) do
				local target = findTowerTarget(tower)
				if target then
					CombatService.TryAutoAttack(tower, target, nil)
				end
				task.wait(0.2)
			end
		end)
	end
end

return TowerService

-- Pré-requis :
-- ServerStorage.JungleMonsters.SmallWolf
-- ServerStorage.JungleMonsters.BigGolem
-- ServerStorage.JungleMonsters.DragonPrototype

-- Workspace.Map.JungleCamps contient des Parts avec attributs :

-- TeamSide = "Neutral" / "Blue" / "Red"

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")

local JungleData = require(ReplicatedStorage.Modules.JungleData)
local CombatService = require(script.Parent.CombatService)
local Utility = require(script.Parent.Utility)

local JungleService = {}

local function setupMonster(monster, campType)
	local data = JungleData.Camps[campType]

	monster:SetAttribute("UnitType", "JungleMonster")
	monster:SetAttribute("Team", "Neutral")
	monster:SetAttribute("CampType", campType)
	monster:SetAttribute("MaxHealth", data.MaxHealth)
	monster:SetAttribute("Health", data.MaxHealth)
	monster:SetAttribute("AttackDamage", data.AttackDamage)
	monster:SetAttribute("AttackRange", data.AttackRange)
	monster:SetAttribute("AttackCooldown", data.AttackCooldown)
	monster:SetAttribute("GoldReward", data.GoldReward)
	monster:SetAttribute("XPReward", data.XPReward)
	monster:SetAttribute("Dead", false)

	local humanoid = monster:FindFirstChildOfClass("Humanoid")
	if humanoid then
		humanoid.MaxHealth = data.MaxHealth
		humanoid.Health = data.MaxHealth
	end

	monster.Parent = workspace.Units
end

local function findNearestEnemyChampion(monster)
	local bestTarget = nil
	local bestDistance = math.huge
	local leashRange = 45

	for _, candidate in ipairs(workspace.Units:GetChildren()) do
		if candidate:GetAttribute("UnitType") == "Champion" and Utility.IsAlive(candidate) then
			local distance = Utility.Distance(monster, candidate)
			if distance <= leashRange and distance < bestDistance then
				bestTarget = candidate
				bestDistance = distance
			end
		end
	end

	return bestTarget
end

local function runMonsterAI(monster, spawnCFrame)
	task.spawn(function()
		while monster and monster.Parent and Utility.IsAlive(monster) do
			local target = findNearestEnemyChampion(monster)

			if target then
				CombatService.TryAutoAttack(monster, target, nil)
			else
				local humanoid = monster:FindFirstChildOfClass("Humanoid")
				if humanoid then
					humanoid:MoveTo(spawnCFrame.Position)
				end
			end

			task.wait(0.3)
		end
	end)
end

function JungleService.SpawnCamp(campPart)
	local campType = campPart:GetAttribute("CampType")
	local data = JungleData.Camps[campType]

	if not data then
		warn("Invalid camp type:", campType)
		return
	end

	local template = ServerStorage.JungleMonsters:FindFirstChild(campType)
	if not template then
		warn("Missing jungle monster template:", campType)
		return
	end

	local monster = template:Clone()
	monster:PivotTo(campPart.CFrame)
	setupMonster(monster, campType)
	runMonsterAI(monster, campPart.CFrame)

	monster:GetAttributeChangedSignal("Dead"):Connect(function()
		if monster:GetAttribute("Dead") == true then
			local respawnTime = data.RespawnTime
			task.delay(respawnTime, function()
				if campPart and campPart.Parent then
					JungleService.SpawnCamp(campPart)
				end
			end)
		end
	end)

-- Contrôle ZQSD.
-- Objectif :
-- - Compatible clavier AZERTY.
-- - Facile à convertir en joystick mobile plus tard.
-- - Pas de QWER pour les sorts maintenant.

local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer

local moveState = {
	Z = false,
	Q = false,
	S = false,
	D = false,
}

local function setKey(input, value)
	local key = input.KeyCode

	if key == Enum.KeyCode.Z then
		moveState.Z = value
	elseif key == Enum.KeyCode.Q then
		moveState.Q = value
	elseif key == Enum.KeyCode.S then
		moveState.S = value
	elseif key == Enum.KeyCode.D then
		moveState.D = value
	end
end

UserInputService.InputBegan:Connect(function(input, gameProcessed)
	if gameProcessed then return end
	setKey(input, true)
end)

UserInputService.InputEnded:Connect(function(input)
	setKey(input, false)
end)

local function getMoveVector()
	local x = 0
	local z = 0

	if moveState.Z then
		z -= 1
	end

	if moveState.S then
		z += 1
	end

	if moveState.Q then
		x -= 1
	end

	if moveState.D then
		x += 1
	end

	local vector = Vector3.new(x, 0, z)

	if vector.Magnitude > 1 then
		vector = vector.Unit
	end

	return vector
end

RunService.RenderStepped:Connect(function()
	local character = player.Character
	if not character then return end

	local humanoid = character:FindFirstChildOfClass("Humanoid")
	if not humanoid then return end

	local vector = getMoveVector()
	humanoid:Move(vector, true)
end)

-- Brouillard de guerre visuel simplifié.
-- Version prototype :
-- - cache les ennemis trop loin du joueur local.
-- Version avancée :
-- - recevoir du serveur la liste exacte des unités visibles.
-- - afficher une texture noire sur la map.
-- - ne pas streamer les positions ennemies invisibles au client si possible.

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local SharedConstants = require(ReplicatedStorage.Modules.SharedConstants)

local player = Players.LocalPlayer

local function setModelVisible(model, visible)
	for _, descendant in ipairs(model:GetDescendants()) do
		if descendant:IsA("BasePart") then
			descendant.LocalTransparencyModifier = visible and 0 or 1
		elseif descendant:IsA("BillboardGui") then
			descendant.Enabled = visible
		end
	end
end

local function getRoot(model)
	return model:FindFirstChild("HumanoidRootPart") or model.PrimaryPart
end

RunService.RenderStepped:Connect(function()
	local character = player.Character
	if not character then return end

	local myRoot = getRoot(character)
	if not myRoot then return end

	for _, unit in ipairs(workspace.Units:GetChildren()) do
		if unit ~= character then
			local unitRoot = getRoot(unit)
			local unitTeam = unit:GetAttribute("Team")
			local myTeam = player:GetAttribute("Team")

			if unitRoot and unitTeam and myTeam then
				if unitTeam == myTeam then
					setModelVisible(unit, true)
				else
					local distance = (myRoot.Position - unitRoot.Position).Magnitude
					local visible = distance <= SharedConstants.VISION.PLAYER_RADIUS
					setModelVisible(unit, visible)
				end
			end
		end
	end
end)

FICHIER : NOTES_IMPORTANTES.txt
================================================================================

1. Ne commence pas par les spells QWER.
   L'auto-attaque + déplacement + vision + minions suffisent déjà pour un gros prototype.

2. Pour Roblox, le serveur doit décider des dégâts.
   Le client ne doit jamais dire : "j'ai fait 500 dégâts".
   Le client doit dire : "je veux attaquer cette cible".

3. Le brouillard de guerre complet est difficile.
   Le prototype peut cacher visuellement les ennemis.
   La version avancée doit éviter d'envoyer trop d'informations au client.

4. Scale énorme :
   Le danger principal est la performance.
   Trop de minions + trop d'IA + trop de checks de distance = lag serveur.
   Il faudra optimiser après la version de base.

5. ZQSD :
   Bon choix pour AZERTY.
   Pour mobile, garde le système abstrait :
   clavier et joystick doivent produire le même MoveVector.

6. Commence avec :
   - 1 champion
   - 3 lanes
   - minions
   - tours
   - jungle simple
   - auto-attaque
   Ensuite seulement : objets, champions multiples, ranked, skins, sorts.
